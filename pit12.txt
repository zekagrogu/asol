using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using System.IO;

namespace kviz
{
    public partial class Form1 : Form
    {
        int brojPitanja = 0;
        int maxPitanja = 10;

        public Form1()
        {
            InitializeComponent();
        }

        private void Form1_Load(object sender, EventArgs e)
        {
            button2.PerformClick();

            // reset
            brojPitanja = 0;
            progressBar1.Value = 0;
            label1.Text = "0";

            button3.PerformClick();
        }

        private void button1_Click(object sender, EventArgs e)
        {
            tabControl1.SelectTab(1);
            button2.PerformClick();
            button3.PerformClick();
        }

        private void button2_Click(object sender, EventArgs e)
        {
            listBox1.Items.Clear();
            listBox1.Items.AddRange(File.ReadAllLines(@"pitanja.txt"));
        }

        private void button3_Click(object sender, EventArgs e)
        {
            int x = listBox1.Items.Count;
            int brpit = x / 5;

            Random z = new Random();
            int p = z.Next(0, brpit);
            p = p * 5;

            label3.Text = listBox1.Items[p].ToString();
            radioButton1.Text = listBox1.Items[p + 1].ToString();
            radioButton2.Text = listBox1.Items[p + 2].ToString();
            radioButton3.Text = listBox1.Items[p + 3].ToString();
            label2.Text = listBox1.Items[p + 4].ToString();

            radioButton1.Checked = false;
            radioButton2.Checked = false;
            radioButton3.Checked = false;
            radioButton4.Checked = false;
        }

        private void button4_Click(object sender, EventArgs e)
        {
            int tacan = Convert.ToInt32(label2.Text);
            int poeni = Convert.ToInt32(label1.Text);

            int izabran = 0;

            if (radioButton1.Checked) izabran = 1;
            else if (radioButton2.Checked) izabran = 2;
            else if (radioButton3.Checked) izabran = 3;
            else if (radioButton4.Checked)
            {
                // "ne znam" → nema promene poena
            }

            // ništa nije izabrano
            if (izabran == 0 && !radioButton4.Checked)
            {
                MessageBox.Show("Izaberi odgovor!");
                return;
            }

            // računanje poena (osim ako je "ne znam")
            if (!radioButton4.Checked)
            {
                if (izabran == tacan)
                    poeni += 10;
                else
                    poeni -= 3;
            }

            label1.Text = poeni.ToString();

            // broj pitanja
            brojPitanja++;

            // progress bar (10 pitanja = 100%)
            progressBar1.Value = brojPitanja * 10;

            // kraj kviza
            if (brojPitanja == maxPitanja)
            {
                MessageBox.Show("Kviz završen! Osvojio si: " + poeni + " poena");

                // reset
                brojPitanja = 0;
                progressBar1.Value = 0;
                label1.Text = "0";

                return;
            }

            // sledeće pitanje
            button3.PerformClick();
        }
    }
}
