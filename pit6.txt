        private void timer2_Tick(object sender, EventArgs e)
        {

            Random x = new Random();
            int pravac = x.Next(0, 4);
            if (pravac == 0) button3.Left -= 2;
            if (pravac == 1) button3.Top += 2;
            if (pravac == 2) button3.Left += 2;
            if (pravac == 3) button3.Top -= 2;

            if (button1.Left > button3.Left) button3.Left += 2;
            if (button1.Left < button3.Left) button3.Left -= 2;
            if (button1.Top > button3.Top) button3.Top += 2;
            if (button1.Top < button3.Top) button3.Top -= 2;
        }
