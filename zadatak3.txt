#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

int main()
{
    int n, k, i, z=0;
    printf("\n Unesi broj n");
    scanf("%d", &n);
    printf("\nUnesi broj k");
    scanf("%d", &k);
    for(i=n;i<k;i++)
        z+=i;
    printf("\n%d",z);
    return 0;
}
