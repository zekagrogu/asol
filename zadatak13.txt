#include <iostream>
#include<stdio.h>
#include<stdlib.h>
#include<time.h>

struct cepanica{
    int duz, pr;
};

using namespace std;

int main()
{
    cepanica niz[1000];
    int i;
    float V, m=0;
    for(i=0;i<1000;i++)
    {
        niz[i].duz=rand()%100;
        niz[i].pr=rand()%100;
        V=niz[i].pr/2*niz[i].pr/2*3.14*niz[i].duz;
        m+=V;
    }
    printf("\nDeda ima %.2f kubika drva", m/1000000);
    return 0;
}
