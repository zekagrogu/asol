#include <stdio.h>

int provera(int x)
{
    if (x < 18) throw 1;
    return printf("Dobrodosli u klub!\n");
}

int main()
{
    int x;
    printf("Unesi koliko imas godina: ");
    scanf("%d", &x);
    try {
        provera(x);
    }
    catch (int p) {
        if (p == 1) printf("Nemate 18 godina. Ulaz u klub nije dozvoljen.\n");
    }
    return 0;
}
